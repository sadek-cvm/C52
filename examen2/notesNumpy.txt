import sys
from PySide6.QtWidgets import QApplication, QWidget, QLabel
from PySide6.QtGui import QPainter, QPixmap, QColor

class SquareDrawer(QWidget):
    def __init__(self):
        super().__init__()

        # Create a QLabel to display the QPixmap
        self.label = QLabel(self)
        self.label.setGeometry(50, 50, 150, 150)  # Set label position and size

        # Draw the square and update the label's pixmap
        self.draw_square()

    def draw_square(self):
        pixmap = QPixmap(200, 200)  # Create a QPixmap with width and height
        pixmap.fill(QColor(255, 255, 255))  # Fill the pixmap with a white background

        painter = QPainter(pixmap)
        painter.setPen(QColor(0, 0, 0))  # Set pen color to black
        painter.setBrush(QColor(255, 0, 0))  # Set brush color to red

        # Draw a square (100x100) at position (50, 50) on the pixmap
        painter.drawRect(50, 50, 100, 100)

        # Set the QPixmap to the label
        self.label.setPixmap(pixmap)

if __name__ == '__main__':
    app = QApplication(sys.argv)
    window = SquareDrawer()
    window.setGeometry(100, 100, 300, 250)  # Set window position and size
    window.setWindowTitle('Square Drawer')

    window.show()
    sys.exit(app.exec())
